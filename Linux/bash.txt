ctrl + u 从游标向前删除指令
ctrl + k 从游标向后删除指令

unset abc 取消变量

echo ${RANDOM} 生成一个随机数，0~32767之间

set bash环境变量，界面变量，自定义变量
  PS1，详情可用man bash查询
    \d 星期，月，日
    \H 完整的主机名称
    \h 主机名称，第一个小数点前面的部分
    \t 24小时，时分秒
    \T 12小时，时分秒
    \u 账号
    \$ root时#，其他$

echo ${$} 变量$保存了当前shell的进程号

echo ${?} 变量?保存了上条命令的执行结果，正常为0

export ABC 将自定义变量转换成环境变量
declare ABC= 将环境变量转换成自定义变量
